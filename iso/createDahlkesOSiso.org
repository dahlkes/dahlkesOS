#+BEGIN_SRC bash
mkdir archlive
cp -r /usr/share/archiso/configs/releng ./archlive
mv ./archlive/releng ./archlive/dahlkesOS
cp dahlkesOS ./archlive/dahlkesOS/airootfs/root/dahlkesOS
mkdir work
sudo mkarchiso -v -w ./work ./archlive/dahlkesOS
rm -r work
rm -r archlive
